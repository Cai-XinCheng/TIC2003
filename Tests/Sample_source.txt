procedure p {
    flag = 0;
    count = 0;
    cenX = 0;
    cenY = 0;
    read x;
    read y;
    while ((x * x + y * y) != 0) {
        count = count + 1;
        cenX = cenX + x;
        cenY = cenY + y;
        read x;
        read y;
    }
    if (count == 0) then {
        flag = 1;
    } else {
        cenX = cenX / count;
        cenY = cenY / count;
    }
    normSq = cenX * cenX + cenY * cenY;
    print flag;
    print cenX;
    print cenY;
    print normSq;
}
